<mx:Application
    xmlns:mx="http://www.adobe.com/2006/mxml"
    xmlns:openzoom="http://ns.openzoom.org/2008"
    viewSourceURL="source/index.html"
    backgroundGradientColors="[ 0, 0 ]"
    layout="absolute"
    applicationComplete="updateConstraints()"
    >
    <mx:Style>
        global {
            theme-color: #222222;
            font-weight: bold;
        }
        CheckBox, Label {
            color: #FFFFFF;
            font-weight: bold;
        }
        Button, NumericStepper {
            fill-alphas: 1, 1, 1, 1;
        }
    </mx:Style>

    <mx:Script>
        <![CDATA[

            import org.openzoom.flash.viewport.constraints.CenterConstraint;
            import org.openzoom.flash.viewport.constraints.CompositeConstraint;
            import org.openzoom.flash.viewport.constraints.VisibilityConstraint;
            import org.openzoom.flash.viewport.constraints.ZoomConstraint;

            private function updateConstraints() : void
            {
                var compositeConstraint : CompositeConstraint =
                                                       new CompositeConstraint()
                var constraints : Array /* of IViewportConstraint */ = []

                if( zoomConstraintCheckBox.selected )
                {
                    var zoomConstraint : ZoomConstraint = new ZoomConstraint()
                    zoomConstraint.minZoom = minZoomNumericStepper.value
                    zoomConstraint.maxZoom = maxZoomNumericStepper.value
                    constraints.push( zoomConstraint )
                }

                if( centerConstraintCheckBox.selected )
                {
                    var centerConstraint : CenterConstraint =
                                                          new CenterConstraint()
                    constraints.push( centerConstraint )
                }

                if( visibilityConstraintCheckBox.selected )
                {
                    var visibilityConstraint : VisibilityConstraint =
                                                      new VisibilityConstraint()
                    constraints.push( visibilityConstraint )
                }

                compositeConstraint.constraints = constraints
                image.constraint = compositeConstraint
            }

        ]]>
    </mx:Script>

    <mx:HBox
        horizontalCenter="0"
        verticalAlign="middle"
        height="36"
        >
        <mx:CheckBox
            id="zoomConstraintCheckBox"
            label="Zoom Contraint"
            selected="true"
            change="updateConstraints()"
            />

        <mx:Label text="Minimum"/>
        <mx:NumericStepper
            id="minZoomNumericStepper"
            enabled="{ zoomConstraintCheckBox.selected }"
            value="0.75"
            minimum="0.01"
            maximum="1"
            stepSize="0.01"
            change="updateConstraints()"
            />

        <mx:Label text="Maximum"/>
        <mx:NumericStepper
            id="maxZoomNumericStepper"
            enabled="{ zoomConstraintCheckBox.selected }"
            value="100"
            minimum="1"
            maximum="2000"
            stepSize="5"
            change="updateConstraints()"
            />

        <mx:Spacer width="12"/>
        <mx:VRule height="18" strokeColor="#444444" strokeWidth="1"/>
        <mx:Spacer width="12"/>

        <mx:CheckBox
            id="centerConstraintCheckBox"
            label="Center Contraint"
            selected="false"
            change="updateConstraints()"
            />

        <mx:Spacer width="12"/>
        <mx:VRule height="18" strokeColor="#444444" strokeWidth="1"/>
        <mx:Spacer width="12"/>

        <mx:CheckBox
            id="visibilityConstraintCheckBox"
            label="Visibility Contraint"
            selected="true"
            change="updateConstraints()"
            />
        <mx:NumericStepper
            enabled="{ visibilityConstraintCheckBox.selected }"
            value="0.5"
            minimum="0"
            maximum="1"
            stepSize="0.05"
            change="updateConstraints()"
            />

        <mx:Spacer width="12"/>
        <mx:VRule height="18" strokeColor="#444444" strokeWidth="1"/>
        <mx:Spacer width="12"/>

        <mx:Button
            label="Show All"
            click="image.showAll()"
            />
    </mx:HBox>
    <openzoom:MultiScaleImage
        id="image"
        top="36" right="0" bottom="0" left="0"
        source="../images/openstreetmap.xml"
        >
        <openzoom:transformer>
            <openzoom:TweenerTransformer/>
        </openzoom:transformer>
        <openzoom:controllers>
            <openzoom:KeyboardController/>
            <openzoom:MouseController/>
        </openzoom:controllers>
    </openzoom:MultiScaleImage>
</mx:Application>
