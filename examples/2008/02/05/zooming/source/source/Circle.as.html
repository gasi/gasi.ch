package
{
    import flash.display.Sprite;


    public class Circle extends Sprite
    {
        public function Circle()
        {
            // draw green circle
            graphics.beginFill( 0x33FF00 )
            graphics.drawCircle( 25, 25, 25 )
            graphics.endFill()
        }
    }
}
