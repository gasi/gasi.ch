<mx:Application
    xmlns:mx="http://www.adobe.com/2006/mxml"
    xmlns:openzoom="http://ns.openzoom.org/2008"
    viewSourceURL="source/index.html"
    backgroundGradientColors="[ 0, 0 ]"
    layout="absolute"
    >
    <mx:Style>
        global {
            theme-color: #222222;
            font-weight: bold;
        }
        ComboBox, Button {
            fill-alphas: 1, 1, 1, 1;
        }
    </mx:Style>

    <mx:Script>
        <![CDATA[

           ;[Bindable]
            private var images : Array =
            [
                { label: "Sound of Silence (Deep Zoom)", source: "../images/sound-of-silence.xml" },
                { label: "No Man Is an Island (Zoomify)", source: "../images/no-man-is-an-island/ImageProperties.xml" },
                { label: "OpenStreetMap", source: "../images/openstreetmap.xml" },
                { label: "Google Maps (Terrain)", source: "../images/google-maps-terrain.xml" },
                { label: "Google Maps (Road)", source: "../images/google-maps-road.xml" },
                { label: "Google Maps (Satellite)", source: "../images/google-maps-satellite.xml" },
                { label: "Flickr", source: "../images/flickr.xml" }
            ]

           ;[Bindable]
            private var source : String = "../images/sound-of-silence.xml"

        ]]>
    </mx:Script>

    <mx:ComboBox
        id="imageComboBox"
        dataProvider="{ images }"
        horizontalCenter="0" y="8"
        change="{ source = imageComboBox.selectedItem.source }"
        />

    <openzoom:MultiScaleImage
        id="image"
        top="40" right="0" bottom="0" left="0"
        source="{ source }"
        zoom="{ zoomSlider.value }"
        >
        <openzoom:controllers>
            <openzoom:KeyboardController/>
            <openzoom:MouseController/>
        </openzoom:controllers>
        <openzoom:transformer>
            <openzoom:TweenerTransformer/>
        </openzoom:transformer>
        <openzoom:constraint>
            <openzoom:CompositeConstraint>
                <openzoom:ZoomConstraint minZoom="0.5" maxZoom="200" />
                <openzoom:VisibilityConstraint visibilityRatio="0.5"/>
                <openzoom:CenterConstraint/>
            </openzoom:CompositeConstraint>
        </openzoom:constraint>
    </openzoom:MultiScaleImage>

    <mx:Button
        label="Show All"
        click="image.showAll()"
         right="8" top="8"/>

    <mx:VSlider
        id="zoomSlider"
        value="1"
        minimum="0.5"
        maximum="100"
        right="10"
        height="400"
        verticalCenter="0"
        liveDragging="true"
        />
</mx:Application>
