<mx:Application
    xmlns:mx="http://www.adobe.com/2006/mxml"
    xmlns:openzoom="http://ns.openzoom.org/2008"
    viewSourceURL="source/index.html"
    backgroundGradientColors="[ 0, 0 ]"
    layout="absolute"
    applicationComplete="updateControllers()"
    >
    <mx:Style>
        global {
            theme-color: #222222;
            font-weight: bold;
        }
        CheckBox, Label {
            color: #FFFFFF;
            font-weight: bold;
        }
        Button, NumericStepper {
            fill-alphas: 1, 1, 1, 1;
        }
    </mx:Style>

    <mx:Script>
        <![CDATA[
            import org.openzoom.flash.viewport.controllers.MouseController;
            import org.openzoom.flash.viewport.controllers.KeyboardController;

            private function updateControllers() : void
            {
                var controllers : Array /* of IViewportController */ = []

                if( mouseControllerCheckBox.selected )
                {
                    var mouseController : MouseController = new MouseController()
                    mouseController.clickZoomInFactor  =
                                           clickZoomInFactorNumericStepper.value
                    mouseController.clickZoomOutFactor =
                                          clickZoomOutFactorNumericStepper.value
                    controllers.push( mouseController )
                }

                if( keyboardControllerCheckBox.selected )
                {
                    var keyboardController : KeyboardController =
                                                        new KeyboardController()

                    keyboardController.upKey     = panUpKeyInput.text
                    keyboardController.downKey   = panDownKeyInput.text
                    keyboardController.leftKey   = panLeftKeyInput.text
                    keyboardController.rightKey  = panRightKeyInput.text
                    keyboardController.panFactor = panFactorNumericStepper.value
                    controllers.push( keyboardController )
                }

                image.controllers = controllers
            }

        ]]>
    </mx:Script>

    <mx:HBox
        horizontalCenter="0"
        verticalAlign="middle"
        height="36"
        >
        <mx:CheckBox
            id="mouseControllerCheckBox"
            label="Mouse Navigation"
            selected="true"
            change="updateControllers()"
            />

        <mx:Label text="Zoom In Factor"/>
        <mx:NumericStepper
            id="clickZoomInFactorNumericStepper"
            enabled="{ mouseControllerCheckBox.selected }"
            value="2"
            minimum="1.1"
            maximum="5.1"
            stepSize="0.2"
            change="updateControllers()"
            />

        <mx:Label text="Zoom Out Factor"/>
        <mx:NumericStepper
            id="clickZoomOutFactorNumericStepper"
            enabled="{ mouseControllerCheckBox.selected }"
            value="0.7"
            minimum="0.2"
            maximum="0.9"
            stepSize="0.05"
            change="updateControllers()"
            />

        <mx:Spacer width="12"/>
        <mx:VRule height="18" strokeColor="#444444" strokeWidth="1"/>
        <mx:Spacer width="12"/>

        <mx:CheckBox
            id="keyboardControllerCheckBox"
            label="Keyboard Navigation"
            selected="false"
            change="updateControllers()"
            />

        <mx:Label text="Up Key"/>
        <mx:TextInput
            id="panUpKeyInput"
            enabled="{ keyboardControllerCheckBox.selected }"
            text="W"
            maxChars="1"
            valueCommit="updateControllers()"
            />

        <mx:Label text="Down Key"/>
        <mx:TextInput
            id="panDownKeyInput"
            enabled="{ keyboardControllerCheckBox.selected }"
            text="S"
            maxChars="1"
            valueCommit="updateControllers()"
            />

        <mx:Label text="Left Key"/>
        <mx:TextInput
            id="panLeftKeyInput"
            enabled="{ keyboardControllerCheckBox.selected }"
            text="A"
            maxChars="1"
            valueCommit="updateControllers()"
            />

        <mx:Label text="Right Key"/>
        <mx:TextInput
            id="panRightKeyInput"
            enabled="{ keyboardControllerCheckBox.selected }"
            text="D"
            maxChars="1"
            valueCommit="updateControllers()"
            />

        <mx:Label text="Pan Factor"/>
        <mx:NumericStepper
            id="panFactorNumericStepper"
            enabled="{ keyboardControllerCheckBox.selected }"
            value="0.1"
            minimum="0.01"
            maximum="0.99"
            stepSize="0.01"
            change="updateControllers()"
            />

        <mx:Spacer width="12"/>
        <mx:VRule height="18" strokeColor="#444444" strokeWidth="1"/>
        <mx:Spacer width="12"/>

        <mx:Button
            label="Show All"
            click="image.showAll()"
            />
    </mx:HBox>
    <openzoom:MultiScaleImage
        id="image"
        top="36" right="0" bottom="0" left="0"
        source="../images/openstreetmap.xml"
        >
        <openzoom:transformer>
            <openzoom:TweenerTransformer/>
        </openzoom:transformer>
        <openzoom:constraint>
            <openzoom:CompositeConstraint>
                <openzoom:VisibilityConstraint/>
                <openzoom:ZoomConstraint minZoom="0.5" maxZoom="2000"/>
            </openzoom:CompositeConstraint>
        </openzoom:constraint>
    </openzoom:MultiScaleImage>
</mx:Application>
